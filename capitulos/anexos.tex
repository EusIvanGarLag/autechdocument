\section*{Anexo A \textemdash{} AnГЎlisis TeГіrico/TГіpico}
\addcontentsline{toc}{section}{Anexo A: AnГЎlisis TeГіrico/TГіpico}

La TeorГ­a de la ComputaciГіn estudia los modelos abstractos de cГіmputo, sus capacidades expresivas
y sus lГ­mites fundamentales. Las ocho unidades de \textbf{AUTECH} recorren la \emph{JerarquГ­a de
Chomsky} en orden ascendente de poder expresivo: desde las cadenas elementales hasta los problemas
que ningГєn algoritmo puede resolver.

% =============================================================================
\subsection*{Unidad 1: Cadenas y Lenguajes вЂ” Fundamentos de la TeorГ­a de SГ­mbolos}
% =============================================================================

La base formal de la computaciГіn es la formalizaciГіn algebraica de la comunicaciГіn entre el ser
humano y la mГЎquina mediante estructuras discretas y bien definidas \cite{hopcroft2007es}.

\subsubsection*{Alfabeto, Cadenas y Operaciones Elementales}

Un \textbf{alfabeto} $\Sigma$ es un conjunto finito y no vacГ­o de sГ­mbolos. La restricciГіn de
finitud garantiza que el procesamiento sea realizable en tiempo y espacio acotados \cite{kelley2001}.
Una \textbf{cadena} $w = a_1 a_2 \cdots a_n$ es una secuencia finita sobre $\Sigma$; su longitud
$|w| = n$ cuenta el nГєmero de posiciones. La \textbf{cadena vacГ­a} $\varepsilon$ tiene
$|\varepsilon| = 0$ y es el elemento identidad de la concatenaciГіn \cite{eugene2008}:

\begin{itemize}
  \item \textbf{ConcatenaciГіn} $uv$: yuxtaposiciГіn de $u$ y $v$; asociativa pero no conmutativa
        \cite{sipser2012}.
  \item \textbf{Potencia} $w^k$: $k$ repeticiones de $w$; $w^0 = \varepsilon$ por convenio.
  \item \textbf{Reversa} $w^R$: imagen especular de $w$; base de los lenguajes de palГ­ndromos
        \cite{decastro2004}.
\end{itemize}

\subsubsection*{Lenguajes y Clausuras}

Un \textbf{lenguaje} $L \subseteq \Sigma^*$ es cualquier conjunto de cadenas. La
\textbf{clausura de Kleene} $\Sigma^* = \bigcup_{i \geq 0}\Sigma^i$ genera todas las cadenas
posibles; la \textbf{clausura positiva} $\Sigma^+ = \Sigma^* \setminus \{\varepsilon\}$ excluye
la vacГ­a \cite{eugene2008}. La validaciГіn $w \in L$ es el precursor formal del analizador lГ©xico
en compiladores \cite{decastro2004}.

\begin{algobox}{ProcesarCadenaGeneral вЂ” Operaciones algebraicas con validaciГіn de alfabeto}
\begin{algorithmic}[1]
  \Require Cadena $w$; alfabeto $\Sigma$;
           operaciГіn $\mathit{op} \in \{\textsc{Long}, \textsc{Inv}, \textsc{Pot}, \textsc{Conc}\}$;
           parГЎmetros $k$ (potencia) o $v$ (concat)
  \Ensure  Resultado de la operaciГіn, o \textsc{Error} si $w \not\subseteq \Sigma$
  \State
  \Function{ValidarAlfabeto}{$w,\,\Sigma$}
    \For{cada sГ­mbolo $a_i \in w$}
      \If{$a_i \notin \Sigma$}
        \State \textbf{lanzar} Error$({\texttt{\textquotedbl{}sГ­mbolo invГЎlido:\textquotedbl{}}} + a_i)$
      \EndIf
    \EndFor
  \EndFunction
  \State
  \State \Call{ValidarAlfabeto}{$w, \Sigma$}
  \State
  \If{$\mathit{op} = \textsc{Long}$}
    \State \Return $|w|$ \Comment{O(1)}
  \ElsIf{$\mathit{op} = \textsc{Inv}$}
    \State $r \gets \varepsilon$
    \For{$i \gets |w|$ \textbf{downto} $1$}
      \State $r \gets r \cdot w[i]$
    \EndFor
    \State \Return $r$ \Comment{O($n$)}
  \ElsIf{$\mathit{op} = \textsc{Pot}$}
    \State $r \gets \varepsilon$
    \For{$j \gets 1$ \textbf{to} $k$}
      \State $r \gets r \cdot w$
    \EndFor
    \State \Return $r$ \Comment{O($k \cdot n$)}
  \ElsIf{$\mathit{op} = \textsc{Conc}$}
    \State \Return $w \cdot v$ \Comment{O($n + |v|$)}
  \EndIf
\end{algorithmic}
\end{algobox}

\subsection*{Unidad 2: AutГіmatas Finitos Deterministas (AFD)}
% =============================================================================

El AFD lee una cadena sГ­mbolo a sГ­mbolo y al concluir acepta o rechaza segГєn el estado en que se
encuentre \cite{sipser2012}. Es el modelo de cГіmputo mГЎs sencillo y su estudio sienta las bases
para todos los modelos superiores.

\subsubsection*{DefiniciГіn Formal (5-tupla) y FunciГіn de TransiciГіn}

Un AFD es una 5-tupla $M = (Q, \Sigma, \delta, q_0, F)$: estados $Q$, alfabeto $\Sigma$,
transiciГіn total $\delta: Q \times \Sigma \to Q$, estado inicial $q_0 \in Q$, estados de
aceptaciГіn $F \subseteq Q$ \cite{hopcroft2007es}. El tГ©rmino \emph{determinista} garantiza que
para cada par $(q, a)$ existe exactamente una transiciГіn, sin ambigГјedad \cite{kelley2001}.

La \textbf{funciГіn extendida} $\hat\delta: Q \times \Sigma^* \to Q$ generaliza $\delta$ a cadenas
completas mediante $\hat\delta(q,\varepsilon)=q$ y $\hat\delta(q, wa) = \delta(\hat\delta(q,w), a)$.
Una cadena $w$ es aceptada si $\hat\delta(q_0, w) \in F$. Los AFD reconocen exactamente la clase
de los \textbf{Lenguajes Regulares}, cerrada bajo uniГіn, intersecciГіn y complemento \cite{hopcroft2007}.

\begin{algobox}{ProcesarCadenaAFD вЂ” SimulaciГіn paso a paso con traza de estados}
\begin{algorithmic}[1]
  \Require AFD $M = (Q,\,\Sigma,\,\delta,\,q_0,\,F)$;\; cadena $w = a_1 a_2 \cdots a_n$
  \Ensure  \textsc{Aceptada}/\textsc{Rechazada} y traza completa de estados visitados
  \State
  \State $q \gets q_0$;\quad traza $\gets \langle q_0 \rangle$
  \For{$i \gets 1$ \textbf{to} $n$}
    \If{$a_i \notin \Sigma$}
      \State \Return \textsc{Error}(\textquotedbl{}sГ­mbolo fuera del alfabeto\textquotedbl{})
    \EndIf
    \State $q \gets \delta(q,\; a_i)$ \Comment{transiciГіn determinista вЂ” O(1)}
    \State traza.agregar$(q)$
    \If{$q$ es estado trampa} \Comment{optimizaciГіn: terminar anticipadamente}
      \State \Return \textsc{Rechazada}, traza
    \EndIf
  \EndFor
  \State
  \If{$q \in F$}
    \State \Return \textsc{Aceptada}, traza
  \Else
    \State \Return \textsc{Rechazada}, traza
  \EndIf
  \State \Comment{Complejidad: O($n$) tiempo;\; O($|Q|\cdot|\Sigma|$) espacio para $\delta$}
\end{algorithmic}
\end{algobox}

\subsection*{Unidad 3: AutГіmatas Finitos No Deterministas (AFN)}
% =============================================================================

El AFN amplГ­a el AFD permitiendo mГєltiples transiciones ante el mismo sГ­mbolo, e incluso
transiciones sin consumir sГ­mbolo alguno ($\varepsilon$-transiciones). El sistema explora todas las
posibilidades en paralelo y acepta si \emph{al menos un hilo} concluye en un estado de aceptaciГіn
\cite{sipser2012}.

\subsubsection*{DefiniciГіn Formal y $\varepsilon$-Cerradura}

Un AFN es $(Q,\Sigma,\delta_N,q_0,F)$ con $\delta_N: Q \times (\Sigma\cup\{\varepsilon\}) \to
\mathcal{P}(Q)$ \cite{eugene2008}. La $\varepsilon$\textbf{-cerradura} $\textsc{EClosure}(S)$ es
el conjunto de todos los estados alcanzables desde $S$ Гєnicamente mediante flechas $\varepsilon$:

\[
  \textsc{EClosure}(S) \;=\; \bigl\{\, q \;\big|\; \exists\, p \in S,\; p \xrightarrow{\,\varepsilon^*\,} q \,\bigr\}
\]

\subsubsection*{Equivalencia AFNвЂ“AFD y MinimizaciГіn}

El \textbf{algoritmo de construcciГіn de subconjuntos} transforma cualquier AFN en un AFD
equivalente: cada estado del AFD representa un subconjunto de estados del AFN, con hasta $2^{|Q|}$
estados en el peor caso \cite{hopcroft2007es}. El AFD mГ­nimo resultante (Гєnico salvo isomorfismo,
por el Teorema de Myhill-Nerode) permite comparar estructuralmente la soluciГіn del alumno con la
soluciГіn canГіnica \cite{kelley2001}.

\begin{algobox}{SimulazAFN вЂ” SimulaciГіn con $\varepsilon$-cerradura y estados activos paralelos}
\begin{algorithmic}[1]
  \Require AFN $N = (Q,\,\Sigma,\,\delta_N,\,q_0,\,F)$;\; cadena $w = a_1 a_2 \cdots a_n$
  \Ensure  \textsc{Aceptado}/\textsc{Rechazado} y evoluciГіn del conjunto de estados activos
  \State
  \Function{EClosure}{$S$} \Comment{BFS sobre $\varepsilon$-transiciones вЂ” O($|Q|+|\delta_N|$)}
    \State $E \gets S$;\quad pila $\gets \text{copiar}(S)$
    \While{pila $\neq \emptyset$}
      \State $q \gets$ pila.desapilar()
      \For{cada $p \in \delta_N(q,\; \varepsilon)$}
        \If{$p \notin E$}
          \State $E \gets E \cup \{p\}$;\quad pila.apilar($p$)
        \EndIf
      \EndFor
    \EndWhile
    \State \Return $E$
  \EndFunction
  \State
  \State $S \gets \Call{EClosure}{\{q_0\}}$ \Comment{InicializaciГіn}
  \State traza $\gets \langle(0,\,S)\rangle$
  \For{$i \gets 1$ \textbf{to} $n$}
    \State $S' \gets \displaystyle\bigcup_{q \in S}\delta_N(q,\; a_i)$ \Comment{Mover con $a_i$}
    \State $S  \gets \Call{EClosure}{S'}$ \Comment{Cerrar bajo $\varepsilon$}
    \State traza.agregar$(i,\, S)$
    \If{$S = \emptyset$}
      \State \textbf{salir} \Comment{Todos los hilos muertos}
    \EndIf
  \EndFor
  \State
  \If{$S \cap F \neq \emptyset$}
    \State \Return \textsc{Aceptado}, traza
  \Else
    \State \Return \textsc{Rechazado}, traza
  \EndIf
\end{algorithmic}
\end{algobox}

\subsection*{Unidad 4: Expresiones Regulares y GramГЎticas Regulares}
% =============================================================================

Las Expresiones Regulares (ER) son la representaciГіn \emph{declarativa} de los Lenguajes
Regulares: donde los autГіmatas describen el comportamiento mediante estados, las ER describen
la estructura mediante patrones algebraicos \cite{sipser2012}.

\subsubsection*{Operadores, Teorema de Kleene y Algoritmo de Thompson}

Tres operadores generan todas las ER sobre $\Sigma$ \cite{kelley2001}: \textbf{uniГіn}
($r_1 \mid r_2$), \textbf{concatenaciГіn} ($r_1 \cdot r_2$) y \textbf{clausura de Kleene} ($r^*$).
El \textbf{Teorema de Kleene} establece la equivalencia exacta entre ER, AFD y AFN. El
\textbf{Algoritmo de Thompson} convierte una ER en un AFN-$\varepsilon$ de forma inductiva y
sistemГЎtica en $O(|r|)$ estados y transiciones \cite{hopcroft2007es}.

\subsubsection*{GramГЎticas Regulares y Lema de Bombeo}

Las \textbf{gramГЎticas lineales por la derecha} ($A \to aB$ Гі $A \to a$) generan exactamente los
Lenguajes Regulares y definen la estructura de los \emph{tokens} en el anГЎlisis lГ©xico
\cite{decastro2004}. El \textbf{Lema de Bombeo} es la herramienta formal para demostrar que
ciertos lenguajes \emph{no} son regulares: si $L$ es regular, $\exists\,p$ tal que toda
$s \in L$ con $|s| \geq p$ se descompone $s = xyz$ con $|xy| \leq p$, $|y| \geq 1$ y
$xy^iz \in L$, $\forall\, i \geq 0$ \cite{sipser2012}.

\newpage
\begin{algobox}{Thompson вЂ” ConstrucciГіn del AFN-$\varepsilon$ por inducciГіn estructural}
\begin{algorithmic}[1]
  \Require ExpresiГіn regular $r$ sobre $\Sigma$
  \Ensure  AFN-$\varepsilon$ $N$ con estado inicial $q_i$ y Гєnico estado final $q_f$
  \State
  \Function{Thompson}{$r$}
    \If{$r = \varepsilon$}
      \State Crear $q_i,\,q_f$;\; aГ±adir $q_i \xrightarrow{\varepsilon} q_f$
      \State \Return $(q_i,\, q_f)$
    \ElsIf{$r = a \in \Sigma$}
      \State Crear $q_i,\,q_f$;\; aГ±adir $q_i \xrightarrow{a} q_f$
      \State \Return $(q_i,\, q_f)$
    \ElsIf{$r = r_1 \mid r_2$} \Comment{UniГіn: bifurcaciГіn $\varepsilon$}
      \State $(i_1,f_1) \gets \Call{Thompson}{r_1}$;\;
             $(i_2,f_2) \gets \Call{Thompson}{r_2}$
      \State Crear $q_i,\,q_f$;\; aГ±adir:
             $q_i\!\to\!i_1,\; q_i\!\to\!i_2,\; f_1\!\to\!q_f,\; f_2\!\to\!q_f$
      \State \Return $(q_i,\, q_f)$
    \ElsIf{$r = r_1 \cdot r_2$} \Comment{ConcatenaciГіn: fusiГіn de estados frontera}
      \State $(i_1,f_1) \gets \Call{Thompson}{r_1}$;\;
             $(i_2,f_2) \gets \Call{Thompson}{r_2}$
      \State Fusionar $f_1 \equiv i_2$
      \State \Return $(i_1,\, f_2)$
    \ElsIf{$r = r_1^*$} \Comment{Kleene: bucle $\varepsilon$ y aceptaciГіn de $\varepsilon$}
      \State $(i_1,f_1) \gets \Call{Thompson}{r_1}$
      \State Crear $q_i,\,q_f$;\; aГ±adir:
             $q_i\!\to\!i_1,\; f_1\!\to\!i_1,\; f_1\!\to\!q_f,\; q_i\!\to\!q_f$
      \State \Return $(q_i,\, q_f)$
    \EndIf
  \EndFunction
  \State
  \State \Return \Call{Thompson}{$r$}
  \State \Comment{Complejidad: O($|r|$) estados y transiciones}
\end{algorithmic}
\end{algobox}

\subsection*{Unidad 5: GramГЎticas Independientes del Contexto (GIC)}
% =============================================================================

Las GIC (Tipo~2 en la JerarquГ­a de Chomsky) describen la mayorГ­a de los lenguajes de programaciГіn
modernos. Son suficientemente expresivas para capturar la recursiГіn y la jerarquГ­a sintГЎctica, pero
suficientemente restringidas para que su reconocimiento sea tratable \cite{sipser2012}.

\subsubsection*{DefiniciГіn Formal (4-tupla), Derivaciones y AmbigГјedad}

Una GIC es $(V,\Sigma,R,S)$: variables $V$, terminales $\Sigma$, reglas $R$ de la forma
$A \to \alpha$ con $A \in V$, $\alpha \in (V \cup \Sigma)^*$, sГ­mbolo inicial $S \in V$
\cite{hopcroft2007es}. Una derivaciГіn \emph{leftmost} sustituye siempre la variable mГЎs a la
izquierda; el \textbf{ГЎrbol de derivaciГіn} visualiza la estructura jerГЎrquica de la cadena
\cite{kelley2001}. Una GIC es \textbf{ambigua} si alguna cadena tiene dos ГЎrboles distintos,
lo cual es indeseable en compiladores pues implica interpretaciones mГєltiples \cite{sipser2012}.

\subsubsection*{Formas Normales}

La \textbf{FNC (Chomsky)} ($A \to BC$ Гі $A \to a$) es la base del algoritmo CYK con complejidad
$O(n^3)$; la \textbf{FNG (Greibach)} ($A \to a\alpha$) facilita la construcciГіn directa de
autГіmatas de pila \cite{hopcroft2007es}.

\begin{algobox}{CYK вЂ” Reconocimiento en GIC mediante programaciГіn dinГЎmica (FNC requerida)}
\begin{algorithmic}[1]
  \Require GIC $G = (V,\,\Sigma,\,R,\,S)$ en \textbf{FNC};\; cadena $w = a_1 \cdots a_n$
  \Ensure  \textsc{SГ­} si $w \in L(G)$, \textsc{No} en otro caso
  \State \Comment{$T[i][j]$ = conjunto de variables que generan $w[i\,{..}\,j]$}
  \State Inicializar tabla $T[1..n][1..n]$ con conjuntos vacГ­os
  \For{$i \gets 1$ \textbf{to} $n$} \Comment{Caso base: subcadenas de longitud 1}
    \For{cada regla $A \to a_i \in R$}
      \State $T[i][i] \gets T[i][i] \cup \{A\}$
    \EndFor
  \EndFor
  \For{$\ell \gets 2$ \textbf{to} $n$} \Comment{Longitud de subcadena creciente}
    \For{$i \gets 1$ \textbf{to} $n - \ell + 1$}
      \State $j \gets i + \ell - 1$
      \For{$k \gets i$ \textbf{to} $j-1$} \Comment{Punto de particiГіn}
        \For{cada regla $A \to BC \in R$}
          \If{$B \in T[i][k]$ \textbf{ y } $C \in T[k+1][j]$}
            \State $T[i][j] \gets T[i][j] \cup \{A\}$
          \EndIf
        \EndFor
      \EndFor
    \EndFor
  \EndFor
  \State
  \If{$S \in T[1][n]$}
    \State \Return \textsc{SГ­} \Comment{Reconstruir ГЎrbol con traza de decisiones}
  \Else
    \State \Return \textsc{No}
  \EndIf
  \State \Comment{Complejidad: O($n^3 \cdot |R|$) tiempo;\; O($n^2 \cdot |V|$) espacio}
\end{algorithmic}
\end{algobox}

\subsection*{Unidad 6: AutГіmatas de Pila (APD)}
% =============================================================================

El APD extiende el AFD con una \textbf{pila} (\emph{stack}) de capacidad ilimitada, permitiendo
reconocer lenguajes con conteo o balanceo ilimitado imposibles para los modelos finitos, como los
parГ©ntesis equilibrados o $\{0^n1^n \mid n \geq 0\}$ \cite{sipser2012}.

\subsubsection*{DefiniciГіn Formal (7-tupla) y Modos de AceptaciГіn}

Un APD es $(Q,\Sigma,\Gamma,\delta,q_0,Z_0,F)$ donde $\Gamma$ es el alfabeto de pila y
$Z_0 \in \Gamma$ su sГ­mbolo inicial \cite{hopcroft2007es}:
\[
  \delta: Q \times (\Sigma \cup \{\varepsilon\}) \times \Gamma \;\longrightarrow\;
  \mathcal{P}\!\left(Q \times \Gamma^*\right)
\]
El APD acepta por \textbf{estado final} ($q \in F$ al terminar) o por \textbf{pila vacГ­a}
(pila completamente vaciada); ambos criterios son equivalentes \cite{hopcroft2007es}. Los APD no
deterministas reconocen exactamente las GIC; los deterministas son adecuados para el anГЎlisis
sintГЎctico de los lenguajes de programaciГіn reales \cite{eugene2008}.

\begin{algobox}{SimulazAPD вЂ” SimulaciГіn por conjunto de configuraciones activas}
\begin{algorithmic}[1]
  \Require APD $M = (Q,\,\Sigma,\,\Gamma,\,\delta,\,q_0,\,Z_0,\,F)$;\;
           cadena $w = a_1\cdots a_n$;\;
           modo $\in \{\text{EstadoFinal},\,\text{PilaVacГ­a}\}$
  \Ensure  \textsc{Aceptada} o \textsc{Rechazada}
  \State \Comment{ConfiguraciГіn: tripleta $(q,\; \text{sufijo no leГ­do},\; \text{contenido pila})$}
  \State $\mathcal{C} \gets \{(q_0,\; w,\; Z_0)\}$
  \While{$\mathcal{C} \neq \emptyset$}
    \State $\mathcal{C}' \gets \emptyset$
    \For{cada $(q,\; a_i{\cdots}a_n,\; \gamma) \in \mathcal{C}$}
      \State \Comment{Transiciones consumiendo $a_i$}
      \For{cada $(q',\beta) \in \delta(q,\; a_i,\; \text{tope}(\gamma))$}
        \State $\mathcal{C}' \gets \mathcal{C}' \cup \{(q',\; a_{i+1}{\cdots}a_n,\; \beta\cdot\text{resto}(\gamma))\}$
      \EndFor
      \State \Comment{Transiciones $\varepsilon$: no consumen entrada}
      \For{cada $(q',\beta) \in \delta(q,\; \varepsilon,\; \text{tope}(\gamma))$}
        \State $\mathcal{C}' \gets \mathcal{C}' \cup \{(q',\; a_i{\cdots}a_n,\; \beta\cdot\text{resto}(\gamma))\}$
      \EndFor
    \EndFor
    \State $\mathcal{C} \gets \mathcal{C}'$
  \EndWhile
  \State
  \State $\mathcal{C}_f \gets$ configuraciones con entrada completamente leГ­da
  \If{modo $=$ EstadoFinal \textbf{ y } $\exists(q,\varepsilon,\gamma)\in\mathcal{C}_f: q\in F$}
    \State \Return \textsc{Aceptada}
  \ElsIf{modo $=$ PilaVacГ­a \textbf{ y } $\exists(q,\varepsilon,\varepsilon)\in\mathcal{C}_f$}
    \State \Return \textsc{Aceptada}
  \Else
    \State \Return \textsc{Rechazada}
  \EndIf
\end{algorithmic}
\end{algobox}

\subsection*{Unidad 7: MГЎquinas de Turing (MT)}
% =============================================================================

Propuesta por Alan Turing en 1936, la MT es el modelo matemГЎtico definitivo de la computaciГіn.
Su memoria de cinta ilimitada le permite simular cualquier proceso algorГ­tmico concebible y
establece los lГ­mites absolutos de lo que puede ser calculado \cite{sipser2012}.

\subsubsection*{Modelo de Cinta y DefiniciГіn Formal (7-tupla)}

La cinta estГЎ dividida en celdas; la cabeza lectora-escritora lee el sГ­mbolo actual, lo
reemplaza y se desplaza una posiciГіn (L Гі R). El \textbf{sГ­mbolo blanco} $B \in \Gamma$ marca
las celdas no escritas \cite{kelley2001}. La 7-tupla $M = (Q,\Sigma,\Gamma,\delta,q_0,B,F)$
con funciГіn de transiciГіn \emph{parcial}:
\[
  \delta: Q \times \Gamma \;\longrightarrow\; Q \times \Gamma \times \{L,\,R\}
\]
Si $\delta(q,a)$ no estГЎ definida, la MT se \textbf{detiene} (\emph{halts}): puede aceptar
entrando en $q_f \in F$ o rechazar implГ­citamente sin definir transiciГіn \cite{hopcroft2007es}.
Notablemente, tambiГ©n puede entrar en \emph{bucle infinito}, distinciГіn crГ­tica en decidibilidad
\cite{eugene2008}. La \textbf{Tesis de Church-Turing} eleva la MT a definiciГіn formal de
В«algoritmoВ»: cualquier proceso computable por cualquier dispositivo fГ­sico puede ser simulado
por una MT \cite{sipser2012}.

\newpage
\begin{algobox}{SimulazMaquinaTuring вЂ” SimulaciГіn completa con detecciГіn de bucle por configuraciГіn}
\begin{algorithmic}[1]
  \Require MT $M = (Q,\,\Sigma,\,\Gamma,\,\delta,\,q_0,\,B,\,F)$;\;
           cadena $w$;\; lГ­mite de pasos $L_{\max}$
  \Ensure  Estado final de la cinta, estado $q$ y diagnГіstico de terminaciГіn
  \State
  \Function{Leer}{cinta, pos}
    \State \Return cinta$[\text{pos}]$ si pos $\in$ cinta, \textbf{si no} $B$
  \EndFunction
  \State
  \State cinta $\gets w$;\; pos $\gets 0$;\; $q \gets q_0$;\; pasos $\gets 0$
  \State historia $\gets \emptyset$ \Comment{Conjunto de configuraciones vistas}
  \While{$q \notin F$ \textbf{ y } pasos $< L_{\max}$}
    \State $a \gets \Call{Leer}{\text{cinta},\, \text{pos}}$
    \State cfg $\gets (q,\, \text{pos},\, \text{hash(cinta)})$
    \If{cfg $\in$ historia}
      \State \Return \textbf{Bucle infinito detectado} \Comment{Comportamiento no decidible}
    \EndIf
    \State historia.agregar(cfg)
    \If{$(q,\, a) \notin \mathrm{dom}(\delta)$}
      \State \Return \textbf{Halt implГ­cito вЂ” Rechazo}, cinta, $q$
    \EndIf
    \State $(q',\, b,\, D) \gets \delta(q,\, a)$
    \State cinta$[\text{pos}] \gets b$;\quad $q \gets q'$
    \State pos $\gets$ pos$+1$ si $D = R$, \textbf{si no} pos$-1$
    \State pasos $\gets$ pasos $+ 1$
  \EndWhile
  \State
  \If{$q \in F$}
    \State \Return \textbf{Aceptada}, cinta, $q$, pasos
  \Else
    \State \Return \textbf{LГ­mite alcanzado вЂ” posible no terminaciГіn}, cinta, $q$
  \EndIf
\end{algorithmic}
\end{algobox}

\subsection*{Unidad 8: Decidibilidad y Computabilidad}
% =============================================================================

La decidibilidad traza la frontera entre los problemas que un algoritmo puede resolver y aquellos
que estГЎn mГЎs allГЎ de toda posibilidad computacional. La expresabilidad matemГЎtica de un problema
no garantiza su computabilidad \cite{sipser2012}.

\subsubsection*{JerarquГ­a de Lenguajes y Problema de la Parada}

Dentro de los lenguajes de Tipo~0 \cite{hopcroft2007es}:
\begin{itemize}
  \item \textbf{Decidibles (recursivos)}: existe una MT decisora que siempre se detiene y
        responde correctamente. Son el modelo ideal del В«algoritmo confiableВ».
  \item \textbf{Turing-reconocibles (recursivamente enumerables)}: la MT acepta las cadenas del
        lenguaje pero puede entrar en bucle ante las que no pertenecen \cite{cooper2003}.
  \item \textbf{co-Turing-reconocibles}: un lenguaje es decidible si y solo si es Turing-reconocible
        y co-Turing-reconocible simultГЎneamente \cite{kelley2001}.
\end{itemize}

El \textbf{Problema de la Parada} pregunta: Вїpuede un programa determinar si cualquier otro
programa termina? Turing demostrГі mediante \textbf{diagonalizaciГіn} que la respuesta es
\emph{no}: tal programa llevarГ­a a una contradicciГіn autorreferencial \cite{sipser2012}. La
\textbf{reducibilidad} permite probar indecidibilidad de nuevos problemas transformГЎndolos en
el Problema de la Parada \cite{decastro2004}.

\subsubsection*{Complejidad: P vs NP}

Entre los problemas decidibles, la complejidad es crucial: \textbf{P} contiene los problemas
solubles en tiempo polinomial; \textbf{NP} los verificables en tiempo polinomial. La pregunta
P $=$ NP es uno de los siete Problemas del Milenio; su resoluciГіn afirmativa colapsarГ­a la
criptografГ­a moderna \cite{eugene2008}.

\newpage
\begin{algobox}{LimpiarGramatica вЂ” EliminaciГіn de sГ­mbolos inГєtiles (dos fases)}
\begin{algorithmic}[1]
  \Require GIC $G = (V,\,\Sigma,\,R,\,S)$
  \Ensure  GramГЎtica equivalente $G'$ sin sГ­mbolos improductivos ni inaccesibles
  \State
  \Function{Productivos}{$G$} \Comment{Fase 1 вЂ” punto fijo ascendente}
    \State $P \gets \{A \in V \;\mid\; \exists\, A\!\to\!w \in R,\; w \in \Sigma^*\}$
    \Repeat
      \For{cada regla $A \to \alpha \in R$}
        \If{todas las variables en $\alpha$ pertenecen a $P$}
          \State $P \gets P \cup \{A\}$
        \EndIf
      \EndFor
    \Until{$P$ no cambia}
    \State \Return $P$
  \EndFunction
  \State
  \Function{Accesibles}{$G$} \Comment{Fase 2 вЂ” BFS desde sГ­mbolo inicial}
    \State $\mathcal{A} \gets \{S\}$;\quad cola $\gets [S]$
    \While{cola $\neq \emptyset$}
      \State $X \gets$ cola.desencolar()
      \For{cada regla $X \to \alpha \in R$}
        \For{cada variable $B$ en $\alpha$}
          \If{$B \notin \mathcal{A}$}
            \State $\mathcal{A} \gets \mathcal{A} \cup \{B\}$;\quad cola.encolar($B$)
          \EndIf
        \EndFor
      \EndFor
    \EndWhile
    \State \Return $\mathcal{A}$
  \EndFunction
  \State
  \State $P \gets \Call{Productivos}{G}$
  \State $G_1 \gets$ remover de $G$ variables $\notin P$ y reglas que las contengan
  \State $\mathcal{A} \gets \Call{Accesibles}{G_1}$
  \State $V' \gets V \cap P \cap \mathcal{A}$;\quad
         $R' \gets \{r \in R \;\mid\; \mathrm{vars}(r) \subseteq V'\}$
  \State \Return $G' = (V',\,\Sigma,\,R',\,S)$
  \State \Comment{Complejidad: O($|V|^2 \cdot |R|$) en el peor caso}
\end{algorithmic}
\end{algobox}
